def removenegatives(data=None, ipfilepath=None):

    if data is None:
        if  ipfilepath is not None:                
            with open(ipfilepath, 'rb') as fd:
                data = pkl.load(fd) 
        else:            
            return None
    
    # ROWS
    
    # This method works but leaves rows that are almost
    # all -1 except for isolated elements containing noise
    # Any not equal to -1 = Not all -1
    # i.e. remove rows with nothing but -1's
    # i.e. assign only rows that are not all -1's to q2
    # remove rows: axis = 1
    # q_pruned_rows = quotes[(quotes != -1).any(axis=1)]            
    # q_pruned_cols = q_pruned_rows.copy()
    
    # remove any row not containing a number of non-Null
    # values >= THRESHOLD, where THRESHOLD = 30% of the
    # number of columns. Only reason for using a percentage
    # is that occasionally a row will have a garbage value
    # the registers as non-NaN and a row that should be
    # deleted is left in the data and wreaks havoc.
    
    THRESHOLD = data.shape[1]
    maxNAcols = int(0.3 * THRESHOLD) 
    
    data = data.replace(-1, np.nan)
    q_pruned_rows = data.dropna(axis=0, thresh = maxNAcols) 
    q_pruned_cols = q_pruned_rows.copy()
    
    # COLUMNS
    
    # This loop works, but is not the best way to do it
    # for x in q_pruned_rows.columns:
    #     if ((q_pruned_rows[x] == -1).any()):
    #         q_pruned_cols = q_pruned_cols.drop(x, 1) 

    # q_pruned_cols = q_pruned_cols.replace(-1, np.nan)
    q_pruned_cols = q_pruned_cols.dropna(axis=1)            
    
    return q_pruned_cols

def do_data_cleansing(X):
    
    h = lambda x: 0 if (
        (np.abs(x) > 5000) or
        (np.isnan(x)) or
        (not np.isfinite(x))
    ) else x

    dfX = pd.DataFrame(X)
    dfXClean = dfX.applymap(h).copy()
    X_Clean = dfXClean.as_matrix().astype(np.float)
    
    return X_Clean

    df['Close'] = -1 
    df = DataFrame(df.stack()) 
    df = df.rename(columns={0: 'stubcol'})

    for ticker in l:

        df1 = df1.rename(columns={0: ticker})
        df = df.join(df1)

    df = df.drop('stubcol', 1)
    
    return df

def retry(f, n_attempts=3):
    "Wrapper function to retry function calls in case of exceptions"
    def wrapper(*args, **kwargs):
        for i in range(n_attempts):
            try:
                return f(*args, **kwargs)
            except Exception:
                if i == n_attempts - 1:
                    raise
    return wrapper

def fetch_quandl_data(symbol, start_date, end_date):
    cols = ["Open", "High", "Low", "Close", "Volume"] 
    # NOTE how the params passed to quandl.get fall 
    # OUTSIDE the parentheses that enclose the arg
    # to retry!!!! The following gives an error:
    #      retry(quandl.get(symbol, 
    #            start_date=start_date, 
    #            end_date=end_date))
    dat = retry(quandl.get)(symbol, 
                start_date=start_date, 
                end_date=end_date)
    dat = dat[cols]    
    if len(dat) > 0:
        idx = date_range(start=start_date, end=end_date)
        # Index must be of type DatetimeIndex 
        # for the reindexing to work.
        dat.index = pd.DatetimeIndex(dat.index)
        dat = dat.reindex(idx, fill_value=-1)    
        # dat['Date'] = dat.index

    # return np.array(dat)
    return dat
